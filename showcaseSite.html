                                            <script>
                                            var codeSection_index_click = 0;
                                                    function codeSection_makeReadable_hover(){
                                                        document.getElementById("codeSection_stickyRevealer_id").classList.add("codeSection_stickyRevealer_transition_hover");
                                                        document.getElementById("codeSection_absoluteRevealer_id").classList.add("codeSection_absoluteRevealer_transition_hover");
                                                    }
                                                    
                                                    
                                                    function codeSection_makeUnReadable_hover(){
                                                        document.getElementById("codeSection_stickyRevealer_id").classList.remove("codeSection_stickyRevealer_transition_hover");
                                                        document.getElementById("codeSection_absoluteRevealer_id").classList.remove("codeSection_absoluteRevealer_transition_hover");
                                                    }
                                            
                                                    function codeSection_focus_click(){
                                                        if(codeSection_index_click == 0){
                                                        document.getElementById("codeSection_stickyRevealer_id").classList.add("codeSection_stickyRevealer_transition_click");
                                                        document.getElementById("codeSection_absoluteRevealer_id").classList.add("codeSection_absoluteRevealer_transition_click");
                                                        codeSection_index_click ++;
                                                        document.getElementById("codeSection_id").scrollIntoView({behavior: "smooth"});
                                                        document.getElementById("codeSection_codeText_id").style["overscroll-behavior"] = "none";
                                                    }
                                                        else {
                                                        document.getElementById("codeSection_stickyRevealer_id").classList.remove("codeSection_stickyRevealer_transition_click");
                                                        document.getElementById("codeSection_absoluteRevealer_id").classList.remove("codeSection_absoluteRevealer_transition_click");
                                                        codeSection_index_click = 0;
                                                        document.getElementById("codeSection_id").scrollIntoView({behavior: "smooth"});
                                                        document.getElementById("codeSection_codeText_id").style["overscroll-behavior"] = "auto";
                                                        
                                                        }
                                                    }
                                                    function codeSection_unFocus(){
                                                        
                                                        document.getElementById("codeSection_stickyRevealer_id").classList.remove("codeSection_stickyRevealer_transition_click");
                                                        document.getElementById("codeSection_absoluteRevealer_id").classList.remove("codeSection_absoluteRevealer_transition_click");
                                                        codeSection_index_click = 0;
                                                        document.getElementById("codeSection_codeText_id").style["overscroll-behavior"] = "auto";
                                                        
                                                    }
                                            </script>
